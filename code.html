function getValues(){
    let sequence=document.getElementById("stringInput").value;
    let processedSequence=processString(sequence);
    let reversedSequence=reverseString(processedSequence);
    let check=checkPalindrome(processedSequence,reversedSequence);
    displayData(check,reversedSequence);
}
//Processes the sequence to remove spaces , capital letters and special characters
function processString(string){
    let processedString = '';
    // detects if there are special characters
    let specialCharacters='/[!@#$%^&*()_+\-=\[\]{};:"\\|,.<>\/?]+/';
    for(let i=0; i<=string.length-1;i++){
        if(!(specialCharacters.includes(string[i]))){
            processedString+=string[i];
        }
    }
    processedString=processedString.toLowerCase();
    return processedString;
}
function reverseString(string){
    let reversedString='';
    for(let i=string.length-1;i>=0;i--){
        reversedString+=string[i];
    }
    return reversedString;

}
function checkPalindrome(string, reversedString){
    if(string==reversedString){
        return true;
    }else{
        return false;
    }

}
function displayData(checkResult,reversedString){
    let jumboTron=document.getElementById("jumbotronOutput");
    let jumboTemplate=document.getElementById("jumboTemplate");
    let jumboElement=document.importNode(jumboTemplate.content,true);
    let jumboText=jumboElement.querySelectorAll("p");
    if(checkResult){
        jumboText[0].textContent=`the string is a palindrome, the reversed string is ${reversedString}`;
    }else{
        jumboText[0].textContent=`the string is not a palindrome, the reversed string is ${reversedString}`;
    }
    jumboTron.appendChild(jumboElement);
    
    
}

The Code is structed in five functions.

getValues()

Input: None
Function:
Collects the string input from the website form and runs the palindrome algorithm
Output:
None

processString(string)

Input: string
Function:
Lower case the string and removes special characters by comparing to a predefined character list
Output:
processedString , lower case and no special characters

reverseString(string)

Input: string, previously processed
Function:
Reverses the string that was previously processed (lower cased and special characters removed)
Output:
Reversed string

checkPalindrome(processedString, reversedString)

Input: processed string , reversed string
Function:
Compares the processed string and reversed string , returns true if they are equal
Output:
True if the strings are equal , false if they are not

displayData()

Input: checkResult , bool ,reversed string
Function:
Utilizes an html template to display the algorithm's result in a jumbotron element
Output:
None
